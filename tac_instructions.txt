FUN Persona::init
	PARAM SELF
	PARAM nombre
	PARAM edad
	LOAD P_0 SELF nombre
	P_0:= nombre
	LOAD P_1 SELF edad
	P_1:= edad
	LOAD P_2 SELF color
	P_2:= "rojo"
ENDFUN
FUN Persona::saludar
	PARAM SELF
	LOAD P_3 SELF color
	T_0:= "Hola, mi nombre es " + P_3
	OUT T_0
ENDFUN
